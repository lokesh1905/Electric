// ── Electric Shop Rate Book — Apps Script ──────────────────────
// SHEETS:
//   "Items" — full data (managed by app, don't edit manually)
//   "Home"  — simple form: fill → Action dropdown → click Add/Update button

const ITEMS_SHEET = 'Items';
const HOME_SHEET  = 'Home';
const LABEL_COL   = 1;
const VALUE_COL   = 2;

// ── Web App entry points ──────────────────────────────────────
function doGet(e) {
  const action = (e.parameter && e.parameter.action) || 'read';
  return respond(handleAction(action, null));
}
function doPost(e) {
  let body = null;
  try { body = JSON.parse(e.postData.contents); } catch(_) {}
  const action = (e.parameter && e.parameter.action) || (body && body.action) || 'write';
  return respond(handleAction(action, body));
}
function respond(obj) {
  const out = ContentService.createTextOutput(JSON.stringify(obj));
  out.setMimeType(ContentService.MimeType.JSON);
  return out;
}

// ── Action router ─────────────────────────────────────────────
function handleAction(action, body) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  try {
    if (action === 'read')       return readItems(ss);
    if (action === 'write')      return writeItems(ss, body);
    if (action === 'addItem')    return addOrUpdateItem(ss, body, false);
    if (action === 'updateItem') return addOrUpdateItem(ss, body, true);
    if (action === 'deleteItem') return deleteItem(ss, body);
    return { error: 'Unknown action: ' + action };
  } catch(err) { return { error: err.message }; }
}

// ── READ all items ────────────────────────────────────────────
function readItems(ss) {
  const sheet = ss.getSheetByName(ITEMS_SHEET);
  if (!sheet) return [];
  const data = sheet.getDataRange().getValues();
  if (data.length <= 1) return [];
  const headers = data[0];
  return data.slice(1).map(row => {
    const obj = {};
    headers.forEach((h, i) => { obj[h] = row[i] === '' ? null : row[i]; });
    try { if (typeof obj.priceHistory === 'string') obj.priceHistory = JSON.parse(obj.priceHistory); } catch(_) { obj.priceHistory = []; }
    return obj;
  });
}

// ── WRITE all items (full replace from app) ───────────────────
function writeItems(ss, body) {
  const items = (body && body.items) ? body.items : [];
  let sheet = ss.getSheetByName(ITEMS_SHEET);
  if (!sheet) sheet = ss.insertSheet(ITEMS_SHEET);
  sheet.clearContents();
  const headers = ['id','name','brand','buyPrice','sellPrice','unit','cat',
                   'note','updated','distributor','location','pinned','priceHistory'];
  sheet.appendRow(headers);
  if (items.length > 0) {
    const rows = items.map(item => headers.map(h => {
      if (h === 'priceHistory') return JSON.stringify(item[h] || []);
      const v = item[h];
      return (v === undefined || v === null) ? '' : v;
    }));
    sheet.getRange(2, 1, rows.length, headers.length).setValues(rows);
  }
  return { ok: true, count: items.length };
}

// ── ADD or UPDATE a single item ───────────────────────────────
function addOrUpdateItem(ss, body, isUpdate) {
  let sheet = ss.getSheetByName(ITEMS_SHEET);
  if (!sheet) sheet = ss.insertSheet(ITEMS_SHEET);
  const headers = ['id','name','brand','buyPrice','sellPrice','unit','cat',
                   'note','updated','distributor','location','pinned','priceHistory'];
  const firstRow = sheet.getRange(1,1,1,headers.length).getValues()[0];
  if (!firstRow[0]) sheet.appendRow(headers);

  const item = body && body.item ? body.item : {};
  const now  = new Date().toLocaleDateString('en-IN',{day:'2-digit',month:'short',year:'numeric'});

  if (isUpdate) {
    const data = sheet.getDataRange().getValues();
    const hdrs = data[0];
    const nameIdx = hdrs.indexOf('name'), idIdx = hdrs.indexOf('id');
    let targetRow = -1;
    for (let i = 1; i < data.length; i++) {
      if ((item.id && String(data[i][idIdx]) === String(item.id)) ||
          (item.name && String(data[i][nameIdx]).toLowerCase() === item.name.toLowerCase())) {
        targetRow = i + 1; break;
      }
    }
    if (targetRow === -1) return { error: 'Item not found: ' + (item.name || item.id) };
    const existingRow = sheet.getRange(targetRow,1,1,hdrs.length).getValues()[0];
    const existing = {};
    hdrs.forEach((h,i) => { existing[h] = existingRow[i]; });
    const updated = headers.map(h => {
      if (h === 'id')     return existing.id;
      if (h === 'pinned') return existing.pinned || false;
      if (h === 'updated') return now;
      if (h === 'priceHistory') {
        let hist = [];
        try { hist = typeof existing.priceHistory === 'string' ? JSON.parse(existing.priceHistory) : (existing.priceHistory || []); } catch(_){}
        const newBuy = parseFloat(item.buyPrice)||0, newSell = parseFloat(item.sellPrice)||0;
        const oldBuy = parseFloat(existing.buyPrice)||0, oldSell = parseFloat(existing.sellPrice)||0;
        if (newBuy !== oldBuy || newSell !== oldSell)
          hist.push({ date:now, buyPrice:newBuy, sellPrice:newSell, type: newBuy > oldBuy ? 'up':'down' });
        return JSON.stringify(hist);
      }
      return item[h] !== undefined && item[h] !== '' ? item[h] : (existing[h] || '');
    });
    sheet.getRange(targetRow,1,1,headers.length).setValues([updated]);
    return { ok:true, action:'updated', name:item.name };
  } else {
    const newId = String(Date.now()) + String(Math.floor(Math.random()*1000));
    const hist  = JSON.stringify([{date:now, buyPrice:parseFloat(item.buyPrice)||0, sellPrice:parseFloat(item.sellPrice)||0, type:'new'}]);
    const row = headers.map(h => {
      if (h === 'id')           return newId;
      if (h === 'updated')      return now;
      if (h === 'pinned')       return false;
      if (h === 'priceHistory') return hist;
      return item[h] !== undefined ? item[h] : '';
    });
    sheet.appendRow(row);
    return { ok:true, action:'added', name:item.name, id:newId };
  }
}

// ── DELETE a single item ──────────────────────────────────────
function deleteItem(ss, body) {
  const sheet = ss.getSheetByName(ITEMS_SHEET);
  if (!sheet) return { error:'Items sheet not found' };
  const item = body && body.item ? body.item : {};
  const data = sheet.getDataRange().getValues();
  const hdrs = data[0];
  const nameIdx = hdrs.indexOf('name'), idIdx = hdrs.indexOf('id');
  for (let i = data.length-1; i >= 1; i--) {
    if ((item.id && String(data[i][idIdx]) === String(item.id)) ||
        (item.name && String(data[i][nameIdx]).toLowerCase() === item.name.toLowerCase())) {
      sheet.deleteRow(i+1);
      return { ok:true, action:'deleted', name:item.name };
    }
  }
  return { error:'Item not found: ' + (item.name || item.id) };
}

// ── SETUP HOME SHEET — run once from editor ───────────────────
function setupHomeSheet() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let home = ss.getSheetByName(HOME_SHEET);
  if (!home) home = ss.insertSheet(HOME_SHEET, 0);
  home.clearContents();
  home.clearFormats();
  home.clearNotes();

  // ── Header banner ─────────────────────────────────────────
  home.getRange(1,1,1,3).merge()
    .setValue('⚡  ELECTRIC SHOP — ADD / UPDATE ITEM')
    .setBackground('#1a1d2e').setFontColor('#ffffff')
    .setFontWeight('bold').setFontSize(13)
    .setHorizontalAlignment('center').setVerticalAlignment('middle');
  home.setRowHeight(1, 44);

  // ── Field rows 2-10 ───────────────────────────────────────
  const fieldLabels = [
    'Item Name *','Brand','Buy Price (₹)','Sell Price (₹)',
    'Unit','Category','Notes / Remarks','Distributor','Location'
  ];
  fieldLabels.forEach((lbl, i) => {
    const row = i + 2;
    home.getRange(row,1).setValue(lbl)
      .setBackground('#2d3561').setFontColor('#c5cae9')
      .setFontWeight('bold').setFontSize(11).setVerticalAlignment('middle');
    home.getRange(row,2)
      .setBackground('#ffffff').setFontSize(11).setVerticalAlignment('middle')
      .setBorder(true,true,true,true,false,false,'#c5cae9',SpreadsheetApp.BorderStyle.SOLID);
    home.setRowHeight(row, 32);
  });

  // ── Dropdowns for Unit (row 6) and Category (row 7) ──────
  const unitRule = SpreadsheetApp.newDataValidation()
    .requireValueInList(['PCS','MTR','KG','BOX','SET','ROLL','PAIR','LOT'], true)
    .setAllowInvalid(true).build();
  home.getRange(6,2).setDataValidation(unitRule).setFontColor('#1a1d2e');

  const catRule = SpreadsheetApp.newDataValidation()
    .requireValueInList(['Wire','Cable','Switch','Socket','MCB','LED','Fan',
                         'Conduit','Junction Box','Panel','Tool','Accessory','Other'], true)
    .setAllowInvalid(true).build();
  home.getRange(7,2).setDataValidation(catRule).setFontColor('#1a1d2e');

  // ── Divider row 11 ────────────────────────────────────────
  home.getRange(11,1,1,3).merge().setBackground('#e8eaf6');
  home.setRowHeight(11, 8);

  // ── Action row 12 ─────────────────────────────────────────
  home.getRange(12,1).setValue('Action')
    .setBackground('#e8eaf6').setFontColor('#3b5bdb')
    .setFontWeight('bold').setFontSize(11).setVerticalAlignment('middle');
  const actionCell = home.getRange(12,2);
  actionCell.setValue('ADD').setBackground('#fff8e1').setFontColor('#e65100')
    .setFontWeight('bold').setFontSize(12).setHorizontalAlignment('center');
  actionCell.setDataValidation(
    SpreadsheetApp.newDataValidation()
      .requireValueInList(['ADD','UPDATE','DELETE'],true).build()
  );
  home.setRowHeight(12, 36);

  // ── Name to Update row 13 ─────────────────────────────────
  home.getRange(13,1).setValue('Name to Update/Delete')
    .setBackground('#e8eaf6').setFontColor('#546e7a')
    .setFontWeight('bold').setFontSize(10).setVerticalAlignment('middle');
  home.getRange(13,2).setBackground('#fafafa').setFontSize(11)
    .setBorder(true,true,true,true,false,false,'#c5cae9',SpreadsheetApp.BorderStyle.SOLID);
  home.setRowHeight(13, 30);

  // ── Status row 14 ────────────────────────────────────────
  home.getRange(14,1).setValue('Status')
    .setBackground('#1b5e20').setFontColor('#ffffff')
    .setFontWeight('bold').setFontSize(11).setVerticalAlignment('middle');
  home.getRange(14,2).setValue('Ready ✅ — fill form and click the Add/Update button')
    .setBackground('#e8f5e9').setFontColor('#2e7d32')
    .setFontWeight('bold').setFontSize(11).setVerticalAlignment('middle');

    .setValue('📋  HOW TO USE')

// ── RUN HOME ACTION — assign to button ───────────────────────
function runHomeAction() {
  const ss   = SpreadsheetApp.getActiveSpreadsheet();
  const home = ss.getSheetByName(HOME_SHEET);
  if (!home) { Browser.msgBox('Home sheet not found. Run setupHomeSheet() first.'); return; }

  // Read all value column cells rows 2-14
  const vals = home.getRange(2, VALUE_COL, 13, 1).getValues();
  const getR = (absRow) => {
    const v = vals[absRow-2][0];
    return (v === null || v === undefined) ? '' : String(v).trim();
  };

  const action     = (getR(12) || 'ADD').toUpperCase();
  const searchName = getR(13);

  const item = {
    name:        getR(2),
    brand:       getR(3),
    buyPrice:    parseFloat(getR(4))  || 0,
    sellPrice:   parseFloat(getR(5)) || 0,
    unit:        getR(6),
    cat:         getR(7),
    note:        getR(8),
    distributor: getR(9),
    location:    getR(10),
  };
  if (!item.name && searchName) item.name = searchName;

  let result;
  if (!item.name) {
    result = { error: 'Item Name is required (row 2)' };
  } else if (action === 'ADD') {
    result = addOrUpdateItem(ss, { item }, false);
  } else if (action === 'UPDATE') {
    result = addOrUpdateItem(ss, { item }, true);
  } else if (action === 'DELETE') {
    result = deleteItem(ss, { item });
  } else {
    result = { error: 'Unknown action: ' + action };
  }

  const statusCell = home.getRange(14, VALUE_COL);
  if (result.ok) {
    const msg = result.action === 'added'   ? '✅ Added: '   + result.name :
                result.action === 'updated' ? '✅ Updated: ' + result.name :
                result.action === 'deleted' ? '🗑️ Deleted: ' + result.name : '✅ Done';
    statusCell.setValue(msg).setBackground('#e8f5e9').setFontColor('#2e7d32');

    // ── CLEAR all input fields after success ─────────────────
    home.getRange(2,  VALUE_COL, 9, 1).clearContent();  // rows 2-10: all fields
    home.getRange(13, VALUE_COL, 1, 1).clearContent();  // row 13: name to update
    // Row 12 (Action) stays as-is so next entry is ready

  } else {
    statusCell.setValue('❌ ' + result.error)
      .setBackground('#fdecea').setFontColor('#c0392b');
  }
  SpreadsheetApp.flush();
}
